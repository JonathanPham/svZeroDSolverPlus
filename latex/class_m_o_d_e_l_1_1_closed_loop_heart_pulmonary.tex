\doxysubsection{Detailed Description}
\subsubsection*{template$<$typename T$>$\newline
class MODEL\+::\+Closed\+Loop\+Heart\+Pulmonary$<$ T $>$}
RCR boundary condition element. 

Models the mechanical behavior of a Windkessel boundary condition.

\[
\begin{circuitikz} \draw
node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
\draw (1,0) node[anchor=south]{$P_{in}$}
to [R, l=$R_p$, *-] (3,0)
to [R, l=$R_d$, *-*] (5,0)
node[anchor=south]{$P_{ref}$}
(3,0) to [C, l=$C$, *-] (3,-1.5)
node[ground]{$P_{C}$};
\end{circuitikz}
\]\hypertarget{class_m_o_d_e_l_1_1_closed_loop_heart_pulmonary_autotoc_md17}{}\doxyparagraph{Governing equations}\label{class_m_o_d_e_l_1_1_closed_loop_heart_pulmonary_autotoc_md17}
\[
R_{d} Q^{e}-P_{c}^{e}+P_{r e f}-R_{d} C \frac{d P_{c}^{e}}{d t}=0
\]

\[
P^{e}-P_{c}^{e}-R_{p} Q^{e}=0
\]\hypertarget{class_m_o_d_e_l_1_1_closed_loop_heart_pulmonary_autotoc_md18}{}\doxyparagraph{Local contributions}\label{class_m_o_d_e_l_1_1_closed_loop_heart_pulmonary_autotoc_md18}
\[
\mathbf{y}^{e}=\left[\begin{array}{lll}P^{e} & Q^{e} & P_{c}^{e}\end{array}\right]^{T}
\]

\[
\mathbf{E}^{e}=\left[\begin{array}{ccc}
0 & 0 & -R_{d} C \\
0 & 0 & 0
\end{array}\right]
\]

\[
\mathbf{F}^{e}=\left[\begin{array}{ccc}
0 & R_{d} & -1 \\
1 & -R_{p} & -1
\end{array}\right]
\]

\[
\mathbf{c}^{e}=\left[\begin{array}{c}
P_{r e f} \\
0
\end{array}\right]
\]
