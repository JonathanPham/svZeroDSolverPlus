15/**
16 * @brief Pressure reference boundary condition.
17 *
18 * Applies a predefined pressure at a boundary.
19 *
20 * \f[
21 * \begin{circuitikz}
22 * \draw (1,0) node[anchor=south]{$P$} to [short, *-] (1.2,0) ;
23 * \draw [-latex] (1.4,0) -- (2.2,0) node[right] {$Q$};
24 * \draw (1,0) to [short, l=, *-] (1,-1)
25 * node[ground]{$\hat{P}$};
26 * \end{circuitikz}
27 * \f]
28 *
29 * ### Governing equations
30 *
31 * \f[
32 * P=\hat{P}
33 * \f]
34 *
35 * ### Local contributions
36 *
37 * \f[
38 * \mathbf{y}^{e}=\left[\begin{array}{ll}P^{e} & Q^{e}\end{array}\right]^{T}
39 * \f]
40 *
41 * \f[
42 * \mathbf{F}^{e}=\left[\begin{array}{ll}1 & 0\end{array}\right]
43 * \f]
44 *
45 * \f[
46 * \mathbf{C}^{e}=\left[\hat{P}\right]
47 * \f]
48 *
49 * @tparam T Scalar type (e.g. `float`, `double`)
50 */
