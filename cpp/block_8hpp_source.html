200  void Block<T>::setup_dofs_(DOFHandler &dofhandler, unsigned int num_equations, unsigned int num_internal_vars)
201  {
202  // Collect external DOFs from inlet nodes
203  for (auto inlet_node : inlet_nodes)
204  {
205  global_var_ids.push_back(inlet_node->pres_dof);
206  global_var_ids.push_back(inlet_node->flow_dof);
207  }
208 
209  // Collect external DOFs from outlet nodes
210  for (auto outlet_node : outlet_nodes)
211  {
212  global_var_ids.push_back(outlet_node->pres_dof);
213  global_var_ids.push_back(outlet_node->flow_dof);
214  }
215 
216  // Register internal variables of block
217  for (unsigned int i = 0; i < num_internal_vars; i++)
218  {
219  global_var_ids.push_back(dofhandler.register_variable("var_" + std::to_string(i) + "_" + name));
220  }
221 
222  // Register equations of block
223  for (unsigned int i = 0; i < num_equations; i++)
224  {
225  global_eqn_ids.push_back(dofhandler.register_equation());
226  }
227  }
