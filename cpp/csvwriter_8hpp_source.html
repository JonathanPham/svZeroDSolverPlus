47/**
48 * @brief Write the solution to a csv file
49 *
50 * @tparam T Scalar type (e.g. `float`, `double`)
51 * @param times Sequence of time steps corresponding to the solutions
52 * @param states Sequence of states corresponding to the time steps
53 * @param model The underlying model
54 * @param mean Toggle whether only the mean over all time steps should be
55 * written
56 * @return CSV encoded output string
57 */
58template <typename T>
59std::string write_csv(std::vector<T> &times,
60 std::vector<ALGEBRA::State<T>> &states,
61 MODEL::Model<T> &model, bool mean = false) {
62 // Create string stream to buffer output
63 std::stringstream out;
64
65 // Write column labels
66 out << "name,time,flow_in,flow_out,pressure_in,pressure_out\n";
67
68 // Create buffer for lines
69 char buff[130];
70
71 // Determine number of time steps
72 T num_steps = times.size();
73
74 unsigned int inflow_dof;
75 unsigned int outflow_dof;
76 unsigned int inpres_dof;
77 unsigned int outpres_dof;
78 for (auto &[key, elem] : model.blocks) {
79 // Extract global solution indices of the block
80 std::string name = "NoName";
81 std::visit(
82 [&](auto &&block) {
83 if (HELPERS::startswith(block.name, "V")) {
84 name = block.name;
85 inflow_dof = block.inlet_nodes[0]->flow_dof;
86 outflow_dof = block.outlet_nodes[0]->flow_dof;
87 inpres_dof = block.inlet_nodes[0]->pres_dof;
88 outpres_dof = block.outlet_nodes[0]->pres_dof;
89 }
90 },
91 elem);
92
93 // Write the solution of the block to the output file
94 if (name != "NoName") {
95 if (mean) {
96 T inflow_mean = 0.0;
97 T outflow_mean = 0.0;
98 T inpres_mean = 0.0;
99 T outpres_mean = 0.0;
100 for (size_t i = 0; i < times.size(); i++) {
101 inflow_mean += states[i].y[inflow_dof];
102 outflow_mean += states[i].y[outflow_dof];
103 inpres_mean += states[i].y[inpres_dof];
104 outpres_mean += states[i].y[outpres_dof];
105 }
106 inflow_mean /= num_steps;
107 outflow_mean /= num_steps;
108 inpres_mean /= num_steps;
109 outpres_mean /= num_steps;
110 sprintf(buff, "%s,,%.16e,%.16e,%.16e,%.16e\n", name.c_str(),
111 inflow_mean, outflow_mean, inpres_mean, outpres_mean);
112 out << buff;
113 } else {
114 for (size_t i = 0; i < times.size(); i++) {
115 sprintf(buff, "%s,%.16e,%.16e,%.16e,%.16e,%.16e\n", name.c_str(),
116 times[i], states[i].y[inflow_dof], states[i].y[outflow_dof],
117 states[i].y[inpres_dof], states[i].y[outpres_dof]);
118 out << buff;
119 }
120 }
121 }
122 }
123
124 return out.str();
125}
