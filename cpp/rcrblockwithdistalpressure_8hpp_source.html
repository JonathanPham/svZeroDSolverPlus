14  /**
15  * @brief RCR boundary condition element
16  *
17  * Models the mechanical behavior of a Windkessel boundary condition.
18  *
19  * \f[
20  * \begin{circuitikz} \draw
21  * node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
22  * \draw (1,0) node[anchor=south]{$P_{in}$}
23  * to [R, l=$R_p$, *-] (3,0)
24  * to [R, l=$R_d$, *-*] (5,0)
25  * node[anchor=south]{$P_{ref}$}
26  * (3,0) to [C, l=$C$, *-] (3,-1.5)
27  * node[ground]{$P_{C}$};
28  * \end{circuitikz}
29  * \f]
30  *
31  * ### Governing equations
32  *
33  * \f[
34  * R_{d} Q^{e}-P_{c}^{e}+P_{r e f}-R_{d} C \frac{d P_{c}^{e}}{d t}=0
35  * \f]
36  *
37  * \f[
38  * P^{e}-P_{c}^{e}-R_{p} Q^{e}=0
39  * \f]
40  *
41  * ### Local contributions
42  *
43  * \f[
44  * \mathbf{y}^{e}=\left[\begin{array}{lll}P^{e} & Q^{e} & P_{c}^{e}\end{array}\right]^{T}
45  * \f]
46  *
47  * \f[
48  * \mathbf{E}^{e}=\left[\begin{array}{ccc}
49  * 0 & 0 & -R_{d} C \\
50  * 0 & 0 & 0
51  * \end{array}\right]
52  * \f]
53  *
54  * \f[
55  * \mathbf{F}^{e}=\left[\begin{array}{ccc}
56  * 0 & R_{d} & -1 \\
57  * 1 & -R_{p} & -1
58  * \end{array}\right]
59  * \f]
60  *
61  * \f[
62  * \mathbf{c}^{e}=\left[\begin{array}{c}
63  * P_{r e f} \\
64  * 0
65  * \end{array}\right]
66  * \f]
67  *
