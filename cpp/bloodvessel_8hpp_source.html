17  /**
18  * @brief Resistor-capacitor-inductor blood vessel with optional stenosis
19  *
20  * Models the mechanical behavior of a bloodvessel with optional stenosis.
21  *
22  * \f[
23  * \begin{circuitikz} \draw
24  * node[left] {$Q_{in}$} [-latex] (0,0) -- (0.8,0);
25  * \draw (1,0) node[anchor=south]{$P_{in}$}
26  * to [R, l=$R$, *-] (3,0)
27  * to [L, l=$L$, *-*] (5,0)
28  * node[anchor=south]{$P_{out}$}
29  * (3,0) to [C, l=$C$, *-] (3,-1.5)
30  * node[ground]{$P_C$};
31  * \draw [-latex] (5.2,0) -- (6,0) node[right] {$Q_{out}$};
32  * \end{circuitikz}
33  * \f]
34  *
35  * ### Governing equations
36  *
37  * \f[
38  * P_{in}^{e}-P_{out}^{e}-R Q_{in}^{e}-L\frac{d Q_{out}^{e}}{dt}=0
39  * \f]
40  *
41  * \f[
42  * Q_{i n}^{e}-Q_{o u t}^{e}-C \frac{d P_{c}^{e}}{d t}=0
43  * \f]
44  *
45  * \f[
46  * P_{i n}^{e}-R Q_{i n}^{e}-P_{c}=0
47  * \f]
48  *
49  * ### Local contributions
50  *
51  * \f[
52  * \mathbf{y}^{e}=\left[\begin{array}{lllll}P_{i n}^{e} & Q_{in}^{e} & P_{out}^{e} & Q_{out}^{e} & P_C\end{array}\right]^{T}
53  * \f]
54  *
55  * \f[
56  * \mathbf{E}^{e}=\left[\begin{array}{ccccc}
57  * 0 & 0 & 0 & -L & 0 \\
58  * 0 & 0 & 0 & 0 & -C \\
59  * 0 & 0 & 0 & 0 & 0
60  * \end{array}\right]
61  * \f]
62  *
63  * \f[
64  * \mathbf{F}^{e}=\left[\begin{array}{ccccc}
65  * 1 & -R_{ste}-R & -1 & 0 & 0 \\
66  * 0 & 1 & 0 & -1 & 0 \\
67  * 1 & -R_{ste}-R & 0 & 0 & -1
68  * \end{array}\right]
69  * \f]
70  *
71  * \f[
72  * \mathbf{D}^{e}=\left[\begin{array}{ccccc}
73  * 0 & -R_{ste} & 0 & 0 & 0 \\
74  * 0 & 0 & 0 & 0 & 0 \\
75  * 0 & -R_{ste} & 0 & 0 & 0
76  * \end{array}\right]
77  * \f]
78  *
79  * with the stenosis resistance \f$ R_{ste}=K_{t} \frac{\rho}{2 A_{o}^{2}}\left(\frac{A_{o}}{A_{s}}-1\right)^{2}|Q_{in}^{e}| \f$.
80  * The constant part of the equation is summarized in \ref Parameters::stenosis_coefficient. \f$R\f$,
81  * \f$C\f$, and \f$L\f$ refer to Poisieuille resistance, capacitance and inductance, respectively.
82  *
83  * @tparam T Scalar type (e.g. `float`, `double`)
84  */
