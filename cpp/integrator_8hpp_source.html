18  /**
19  * @brief Generalized-alpha integrator
20  *
21  * This class handles the time integration scheme for solving 0D blood
22  * flow system.
23  *
24  * Flow rate, pressure, and other hemodynamic quantities in 0D models of
25  * vascular anatomies are governed by a system of nonlinear differential-algebraic equations (DAEs):
26  *
27  * \f[
28  * \mathbf{E}(\mathbf{y}, t) \cdot \dot{\mathbf{y}}+\mathbf{F}(\mathbf{y}, t) \cdot \mathbf{y}+\mathbf{c}(\mathbf{y}, t)=\mathbf{0}
29  * \f]
30  *
31  * Here, \f$y\f$ is the vector of solution quantities and \f$\dot{y}\f$ is the time derivative of \f$y\f$.
32  * \f$N\f$ is the total number of equations and the total number of global unknowns. The DAE
33  * system is solved implicitly using the generalized-\f$\alpha\f$ method \cite JANSEN2000305.
34  *
35  * @tparam T Scalar type (e.g. `float`, `double`)
36  * @tparam S Type of System (e.g. `ALGEBRA::DenseSystem`, `ALGEBRA::SparseSystem`)
37  */

88  template <typename T, template <class> class S>
89  Integrator<T, S>::Integrator(S<T> &system, T time_step_size, T rho, T atol, int max_iter)
90  {
91  alpha_m = 0.5 * (3.0 - rho) / (1.0 + rho);
92  alpha_f = 1.0 / (1.0 + rho);
93  alpha_m_inv = alpha_m / 1.0;
94  alpha_f_inv = alpha_f / 1.0;
95  gamma = 0.5 + alpha_m - alpha_f;
96  gamma_inv = 1.0 / gamma;
97 
98  this->system = system;
99  this->time_step_size = time_step_size;
100  this->atol = atol;
101  this->max_iter = max_iter;
102  time_step_size_inv = 1.0 / time_step_size;
103 
104  size = system.F.row(0).size();
105  y_af = Eigen::Matrix<T, Eigen::Dynamic, 1>(size);
106  ydot_am = Eigen::Matrix<T, Eigen::Dynamic, 1>(size);
107 
108  y_dot_coeff = alpha_m / (alpha_f * gamma) * time_step_size_inv;
109  }

116  template <typename T, template <class> class S>
117  State<T> Integrator<T, S>::step(State<T> &state, T time, MODEL::Model<T> &model)
118  {
119 
120  // Create new state
121  State<T> old_state = state;
122  State<T> new_state;
123 
124  // Predictor step
125  new_state.y = old_state.y + 0.5 * time_step_size * old_state.ydot;
126  new_state.ydot = old_state.ydot * (gamma - 0.5) * gamma_inv;
127 
128  // Initiator step
129  y_af = old_state.y + alpha_f * (new_state.y - old_state.y);
130  ydot_am = old_state.ydot + alpha_m * (new_state.ydot - old_state.ydot);
131 
132  // Determine new time
133  T new_time = time + alpha_f * time_step_size;
134 
135  // Update time-dependent element contributions in system
136  model.update_time(system, new_time);
137 
138  for (size_t i = 0; i < max_iter; i++)
139  {
140  // Update solution-dependent element contribitions
141  model.update_solution(system, y_af);
142 
143  // Update residuum and check termination criteria
144  system.update_residual(y_af, ydot_am);
145  if (system.residual.cwiseAbs().maxCoeff() < atol)
146  {
147  break;
148  }
149 
150  // Abort if maximum number of non-linear iterations is reached
151  else if (i == max_iter - 1)
152  {
153  throw std::runtime_error("Maxium number of non-linear iterations reached.");
154  }
155 
156  // Determine jacobian
157  system.update_jacobian(y_dot_coeff);
158 
159  // Solve system
160  system.solve();
161 
162  // Add increment to solution
163  y_af += system.dy;
164  ydot_am += system.dy * y_dot_coeff;
165  }
166 
167  // Set new state
168  new_state.y = old_state.y + (y_af - old_state.y) * alpha_f_inv;
169  new_state.ydot = old_state.ydot + (ydot_am - old_state.ydot) / alpha_m_inv;
170 
171  return new_state;
172  }
