18/**
19 * @brief Write the solution to a json file
20 *
21 * @tparam T Scalar type (e.g. `float`, `double`)
22 * @param times Sequence of time steps corresponding to the solutions
23 * @param states Sequence of states corresponding to the time steps
24 * @param model The underlying model
25 * @return Json encoded output string
26 */
27template <typename T>
28std::string write_json(std::vector<T> &times,
29 std::vector<ALGEBRA::State<T>> &states,
30 MODEL::Model<T> &model) {
31 Json::Value output;
32 Json::Value json_times(Json::arrayValue);
33 for (auto time : times) {
34 json_times.append(Json::Value(time));
35 }
36
37 Json::Value json_names(Json::arrayValue);
38 Json::Value json_flow_in(Json::arrayValue);
39 Json::Value json_flow_out(Json::arrayValue);
40 Json::Value json_pres_in(Json::arrayValue);
41 Json::Value json_pres_out(Json::arrayValue);
42
43 for (auto &[key, elem] : model.blocks) {
44 std::string name = "NoName";
45 unsigned int inflow_dof;
46 unsigned int outflow_dof;
47 unsigned int inpres_dof;
48 unsigned int outpres_dof;
49 std::visit(
50 [&](auto &&block) {
51 if (HELPERS::startswith(block.name, "V")) {
52 name = block.name;
53 inflow_dof = block.inlet_nodes[0]->flow_dof;
54 outflow_dof = block.outlet_nodes[0]->flow_dof;
55 inpres_dof = block.inlet_nodes[0]->pres_dof;
56 outpres_dof = block.outlet_nodes[0]->pres_dof;
57 }
58 },
59 elem);
60
61 if (name != "NoName") {
62 json_names.append(name);
63 Json::Value json_flow_in_i(Json::arrayValue);
64 Json::Value json_flow_out_i(Json::arrayValue);
65 Json::Value json_pres_in_i(Json::arrayValue);
66 Json::Value json_pres_out_i(Json::arrayValue);
67 for (auto state : states) {
68 json_flow_in_i.append(state.y[inflow_dof]);
69 json_flow_out_i.append(state.y[outflow_dof]);
70 json_pres_in_i.append(state.y[inpres_dof]);
71 json_pres_out_i.append(state.y[outpres_dof]);
72 }
73 json_flow_in.append(json_flow_in_i);
74 json_flow_out.append(json_flow_out_i);
75 json_pres_in.append(json_pres_in_i);
76 json_pres_out.append(json_pres_out_i);
77 }
78 }
79
80 output["time"] = json_times;
81 output["names"] = json_names;
82 output["flow_in"] = json_flow_in;
83 output["flow_out"] = json_flow_out;
84 output["pressure_in"] = json_pres_in;
85 output["pressure_out"] = json_pres_out;
86
87 Json::FastWriter writer;
88 return writer.write(output);
89}
